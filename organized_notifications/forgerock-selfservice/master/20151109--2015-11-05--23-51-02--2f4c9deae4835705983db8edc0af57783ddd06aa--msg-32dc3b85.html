78 +     @Override

107 +         System.arraycopy(saltBytes, 0, hashPlusSalt, digestBytes.length, NUM_SALT_BYTES);

110 +     }

115 +         byte[] saltBytes;
116 +         byte[] digestBytes = new byte[digestSize];

130 +         } catch (Exception e) {

132 +             return false;
133 +         }

136 +         int plainBytesLength = plaintextfield.length();
137 +         byte[] plainPlusSalt = new byte[plainBytesLength + saltLength];
138 +

140 +         System.arraycopy(saltBytes, 0, plainPlusSalt, plainBytesLength, saltLength);
141 +
142 +         byte[] userDigestBytes;
143 +
