+            }

+            } finally {
+                stream.close();
+            }
+            break;
+        }
+    }

+     * @throws IIOInvalidTreeException Failed to write metadata.

+            throws IIOInvalidTreeException {

+        final double dotsPerMillimeter = dotsPerInch * inchesPerMillimeter;
+
+        IIOMetadataNode horizontalPixelSize = new IIOMetadataNode(&quot;HorizontalPixelSize&quot;);
+        horizontalPixelSize.setAttribute(&quot;value&quot;, Double.toString(dotsPerMillimeter));
+
+        IIOMetadataNode verticalPixelSize = new IIOMetadataNode(&quot;VerticalPixelSize&quot;);
+        verticalPixelSize.setAttribute(&quot;value&quot;, Double.toString(dotsPerMillimeter));
+
+        IIOMetadataNode dimension = new IIOMetadataNode(&quot;Dimension&quot;);
+        dimension.appendChild(horizontalPixelSize);
+        dimension.appendChild(verticalPixelSize);
+
+        IIOMetadataNode root = new IIOMetadataNode(&quot;javax_imageio_1.0&quot;);
+        root.appendChild(dimension);
+
+        metadata.mergeTree(&quot;javax_imageio_1.0&quot;, root);
+    }
